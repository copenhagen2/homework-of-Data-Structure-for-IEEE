int main()
{
    char pre[27], mid[27];
    char con[33];
    for (int i = 0; i < 32; i++) con[i] = '@';
    con[32] = 0;
    cin >> pre;
    cin >> mid;
    int n = strlen(pre);
    traverse2mer(pre, mid, n, con, 0);
    int i = 0;
    while(n) {
        if (con[i] == '@') cout << "NULL" << ' ';
        else {
            cout << con[i] << ' ';
            n--;
        }
        i++;
    }
    return 0;
}

void traverse2mer(char *pre, char *mid, int size, char *mer, int index) {
    if (!size) return;
    char root = *pre;
    int c = 0;
    while(mid[c] != root) c++;
    int leftSize = c;
    int rightSize = size - 1 - c;
    mer[index] = root;
    traverse2mer(pre+1, mid, leftSize, mer, 2*index+1);
    traverse2mer(pre+leftSize+1, mid+leftSize+1, rightSize, mer, 2*index+2);
}
